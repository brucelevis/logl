# the sokol implementations library 
fips_begin_lib(sokol)
    fips_vs_warning_level(4)
    if (FIPS_OSX)
        fips_files(sokol.m)
        if (FIPS_IOS)
            if (SOKOL_USE_METAL)
                fips_frameworks_osx(UIKit Metal MetalKit)
            else()
                fips_frameworks_osx(UIKit OpenGLES GLKit)
            endif()
        else()
            if (SOKOL_USE_METAL)
                fips_frameworks_osx(Cocoa QuartzCore Metal MetalKit)
            else()
                fips_frameworks_osx(Cocoa QuartzCore OpenGL)
            endif()
        endif()
    else()
        fips_files(sokol.c)
        if (FIPS_ANDROID)
            fips_libs(GLESv3 EGL OpenSLES log android)
        elseif (FIPS_LINUX)
            fips_libs(X11 Xi Xcursor GL m dl)
        endif()
    endif()
fips_end_lib()
